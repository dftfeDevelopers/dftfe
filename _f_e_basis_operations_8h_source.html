32 enum UpdateFlags
33 {
34 update_default = 0,
35
36 update_values = 0x0001,
37
38 update_gradients = 0x0002,
39
40 update_transpose = 0x0004,
41
42 update_quadpoints = 0x0008,
43
44 update_inversejacobians = 0x0010,
45
46 update_jxw = 0x0020,
47 };
48
49 inline UpdateFlags
50 operator|(const UpdateFlags f1, const UpdateFlags f2)
51 {
52 return static_cast<UpdateFlags>(static_cast<unsigned int>(f1) |
53 static_cast<unsigned int>(f2));
54 }
55
56
57
58 inline UpdateFlags &
59 operator|=(UpdateFlags &f1, const UpdateFlags f2)
60 {
61 f1 = f1 | f2;
62 return f1;
63 }
64
65
66 inline UpdateFlags operator&(const UpdateFlags f1, const UpdateFlags f2)
67 {
68 return static_cast<UpdateFlags>(static_cast<unsigned int>(f1) &
69 static_cast<unsigned int>(f2));
70 }
71
72
73 inline UpdateFlags &
74 operator&=(UpdateFlags &f1, const UpdateFlags f2)
75 {
76 f1 = f1 & f2;
77 return f1;
78 }

306 /**
307 * @brief Shape function values at quadrature points in ValueTypeBasisData.
308 * @param[in] transpose if false the the data is indexed as [iQuad *
309 * d_nDofsPerCell + iNode] and if true it is indexed as [iNode *
310 * d_nQuadsPerCell + iQuad].
311 */
312 const auto &
313 shapeFunctionBasisData(bool transpose = false) const
314 {
315 if constexpr (std::is_same<ValueTypeBasisCoeff,
316 ValueTypeBasisData>::value)
317 {
318 return transpose ?
319 d_shapeFunctionDataTranspose.find(d_quadratureID)->second :
320 d_shapeFunctionData.find(d_quadratureID)->second;
321 }
322 else
323 {
324 return transpose ?
325 d_shapeFunctionBasisDataTranspose.find(d_quadratureID)
326 ->second :
327 d_shapeFunctionBasisData.find(d_quadratureID)->second;
328 }
329 }
330 /**
331 * @brief Shape function gradient values at quadrature points in ValueTypeBasisData.
332 * @param[in] transpose if false the the data is indexed as [iDim *
333 * d_nQuadsPerCell * d_nDofsPerCell + iQuad * d_nDofsPerCell + iNode] and
334 * if true it is indexed as [iDim * d_nQuadsPerCell * d_nDofsPerCell +
335 * iNode * d_nQuadsPerCell + iQuad].
336 */
337 const auto &
338 shapeFunctionGradientBasisData(bool transpose = false) const
339 {
340 if constexpr (std::is_same<ValueTypeBasisCoeff,
341 ValueTypeBasisData>::value)
342 {
343 return transpose ?
344 d_shapeFunctionGradientDataTranspose.find(d_quadratureID)
345 ->second :
346 d_shapeFunctionGradientData.find(d_quadratureID)->second;
347 }
348 else
349 {
350 return transpose ?
351 d_shapeFunctionGradientBasisDataTranspose
352 .find(d_quadratureID)
353 ->second :
354 d_shapeFunctionGradientBasisData.find(d_quadratureID)
355 ->second;
356 }
357 }

359 /**
360 * @brief Inverse Jacobian matrices in ValueTypeBasisData, for cartesian cells returns the
361 * diagonal elements of the inverse Jacobian matrices for each cell, for
362 * affine cells returns the 3x3 inverse Jacobians for each cell otherwise
363 * returns the 3x3 inverse Jacobians at each quad point for each cell.
364 */
365 const auto &

367 {

377 return d_inverseJacobianBasisData

379 ->second;
